const { voronoi, noise, shape, o0, mouse, src, width, height } = synth voronoi(99,0.9,9) .rotate(12,.3) .sub(noise(20,0.03).luma(0.6)) .modulateKaleid(voronoi(0.6,0.9,0.9)) .colorama(0.1) .repeat(3,3) .modulateRotate(o0) .color(1,0.6,0.2) .add(shape(3,0.04,0.01) .modulateScale(src(o0). modulateRepeat(voronoi(2,0.9,99)))) .repeat(3,3) .kaleid(1) .kaleid(99) .rotate(3,0.1) .sub(src(o0).modulateScale(noise(2,0.2)),0.9) .scale(()=>mouse.y*-2/height-0.1,()=>mouse.x*-2/width-0.1) .mult(shape(99,0.01,0.01).scale(()=>mouse.y*-1.7/height-0.1,()=>mouse.x*-1.7/width-0.1).invert()) .brightness(0.2) .saturate(1) .contrast(1.8) .rotate(3,-.05) .blend(o0,0.98) .scale(1.007, 0.993) .out()
